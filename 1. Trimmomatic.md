Trimmomatic processes raw reads to improve quality before downstream analysis (we do it on the fastq.qz files)
remove adapter sequences, cut low-quality bases, remove very short reads, trim poor-quality ends.
This improves mapping/alignment, assembly quality, expression analysis and variant calling.

```
#PBS -N trimmomatic_adapterclip_limenitis
#PBS -l select=1:ncpus=1:mem=50gb:scratch_local=50gb
#PBS -l walltime=01:00:00

module add trimmomatic

    trimmomatic PE -threads 1 /storage/brno2/home/kat/Lim3_fastq/Lim3_SP011_MKDL250017427-1A_233YGMLT4_L6_1.fq.gz \
 	/storage/brno2/home/kat/Lim3_fastq/Lim3_SP011_MKDL250017427-1A_233YGMLT4_L6_2.fq.gz \
        /storage/brno2/home/kat/Lim3_fastq/trimmomatic/Lim3_SP011_MKDL250017427-1A_233YGMLT4_L6_1.paired.fq.gz \
	/storage/brno2/home/kat/Lim3_fastq/trimmomatic/Lim3_SP011_MKDL250017427-1A_233YGMLT4_L6_2.paired.fq.gz \
	/storage/brno2/home/kat/Lim3_fastq/trimmomatic/Lim3_SP011_MKDL250017427-1A_233YGMLT4_L6_1.unpaired.fq.gz \
	/storage/brno2/home/kat/Lim3_fastq/trimmomatic/Lim3_SP011_MKDL250017427-1A_233YGMLT4_L6_2.unpaired.fq.gz \

        ILLUMINACLIP:/storage/brno2/home/kat/trimmomatic/adapters.fas:2:30:10 \
        MINLEN:80

```
